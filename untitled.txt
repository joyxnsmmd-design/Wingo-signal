<script>
function predict() {
    let period = document.getElementById("periodNumber").value;
    let current = document.getElementById("currentNumber").value;

    if (period.length !== 4 || current === "") {
        document.getElementById("result").innerText = "দয়া করে সঠিক ইনপুট দিন";
        return;
    }

    // কাস্টম ডেমো ফর্মুলা (মজা করার জন্য)
    let sum = parseInt(period.split('').reduce((a, b) => parseInt(a) + parseInt(b))) + parseInt(current);
    let prediction = sum % 10;

    let color = (prediction % 2 === 0) ? "Green" : "Red";

    document.getElementById("result").innerHTML = 
        `পরবর্তী সংখ্যা: <span style="color:yellow">${prediction}</span> <br> সম্ভাব্য রঙ: <span style="color:${color.toLowerCase()}">${color}</span>`;
}
</script>
